Create Table TipoProduto(
[Id_TipoProduto] INTEGER IDENTITY(1,1) PRIMARY KEY,
[Desc_TipoProduto] VARCHAR(75) NOT NULL,
[Ativo] BIT NOT NULL
)
GO
Create Table Produto(
[Id_Produto] INTEGER IDENTITY(1,1) PRIMARY KEY,
[Nome_Produto] VARCHAR(75) NOT NULL,
[Desc_Produto] VARCHAR(100) NOT NULL,
[Vali_Produto] VARCHAR(100) NOT NULL,
[Ativo] BIT NOT NULL
)
GO
Create Table Sabor(
[Id_Sabor] INTEGER IDENTITY(1,1) PRIMARY KEY,
[Desc_Sabor] VARCHAR(100) NOT NULL,
[Ativo] BIT NOT NULL
)
GO
Create Table Preco(
[Id_Preco] INTEGER IDENTITY(1,1) PRIMARY KEY,
[Valor_Produto] FLOAT NOT NULL
)
GO
Create Table TipoProdutoXProdutoXSaborXPreco(
[Id_TipoProdutoXPreco] INTEGER IDENTITY(1,1) PRIMARY KEY,
[Id_TipoProduto] INTEGER NOT NULL,
[Id_Produto] INTEGER NOT NULL,
[Id_Sabor] INTEGER NOT NULL,
[Id_Preco] INTEGER NOT NULL,
FOREIGN KEY ([Id_TipoProduto]) REFERENCES TipoProduto([Id_TipoProduto]),
FOREIGN KEY ([Id_Produto]) REFERENCES Produto([Id_Produto]),
FOREIGN KEY ([Id_Sabor]) REFERENCES Sabor([Id_Sabor]),
FOREIGN KEY ([Id_Preco]) REFERENCES Preco([Id_Preco])
)
GO
